    <script>
        const canvas = document.getElementById('fireworks');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        // ç”Ÿæˆéšæœºæ•°
        function random(min, max) {
            return Math.random() * (max - min) + min;
        }

        // çƒŸèŠ±ç±»
        class Firework {
            constructor(x = null, y = null) {
                // å¦‚æœæœ‰ä¼ å…¥åæ ‡ï¼Œå°±ç”¨ä¼ å…¥çš„ï¼›å¦åˆ™éšæœºç”Ÿæˆ
                this.x = x || random(canvas.width * 0.1, canvas.width * 0.9);
                this.y = y || canvas.height;
                this.speedY = random(-8, -4);
                this.speedX = random(-2, 2);
                this.hue = random(0, 360);
                this.particles = [];
                this.particleCount = 50;
                this.exploded = false;
                this.targetHeight = y ? y : random(canvas.height * 0.1, canvas.height * 0.4);
            }

            update() {
                if (!this.exploded) {
                    this.y += this.speedY;
                    this.x += this.speedX;
                    this.speedY += 0.05;
                    if (this.y <= this.targetHeight) {
                        this.explode();
                    }
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, 2, 0, Math.PI * 2);
                    ctx.fillStyle = `hsl(${this.hue}, 100%, 50%)`;
                    ctx.fill();
                } else {
                    this.particles.forEach(particle => particle.update());
                }
            }

            explode() {
                this.exploded = true;
                for (let i = 0; i < this.particleCount; i++) {
                    this.particles.push(new Particle(this.x, this.y, this.hue));
                }
            }
        }

        // çˆ†ç‚¸ç²’å­ç±»
        class Particle {
            constructor(x, y, hue) {
                this.x = x;
                this.y = y;
                this.speedX = random(-5, 5);
                this.speedY = random(-5, 5);
                this.hue = hue;
                this.alpha = 1;
                this.decay = random(0.01, 0.03);
                this.gravity = 0.05;
            }

            update() {
                this.speedY += this.gravity;
                this.x += this.speedX;
                this.y += this.speedY;
                this.alpha -= this.decay;
                ctx.beginPath();
                ctx.arc(this.x, this.y, 2, 0, Math.PI * 2);
                ctx.fillStyle = `hsla(${this.hue}, 100%, 50%, ${this.alpha})`;
                ctx.fill();
            }
        }

        let fireworks = [];

        // åˆ›å»ºæ–°çƒŸèŠ±
        function createFirework(x, y) {
            fireworks.push(new Firework(x, y));
        }

        // åŠ¨ç”»å¾ªç¯
        function animate() {
            ctx.fillStyle = 'rgba(0, 0, 0, 0.1)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            fireworks.forEach((firework, index) => {
                firework.update();
                if (firework.exploded && firework.particles.every(p => p.alpha <= 0)) {
                    fireworks.splice(index, 1);
                }
            });
            requestAnimationFrame(animate);
        }

        // è‡ªåŠ¨ç”ŸæˆçƒŸèŠ±
        setInterval(() => createFirework(), 1500);
        animate();

        // ç‚¹å‡»å±å¹•ç”ŸæˆçƒŸèŠ±
        canvas.addEventListener('click', (e) => {
            const rect = canvas.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;
            // åœ¨ç‚¹å‡»ä½ç½®ç›´æ¥çˆ†ç‚¸
            createFirework(x, y);
        });

        // é€‚é…çª—å£å¤§å°
        window.addEventListener('resize', () => {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        });
    </script>
